here 2 is operand and + is operator

  1. Arithmatic operator (+,-,*,/,%,floor(),ceil())
    Addition (+)
    console.log(10+20);
    Output: 30

    Subtraction (-)
    console.log(10-20);
    Output: -10

    Multiplication (*)
    console.log(5*10);
    Output: 50

    Division (/)
    console.log(10/5);
    Output: 2

    Modulus (%)
    console.log(11%5);
    Output: 1 (remainder)

    Floor function (Math.floor())
    console.log(Math.floor(11/5)); (where math is a file and floor is a function)
    Output: 2 (floor value of 2.2)

    Ceil function (Math.ceil())
    console.log(Math.ceil(11/5)); (where math is a file and ceil is a function)
    Output: 3 (ceil or next value of 2.2)
  1. Relational operator (<,>,<=,>=,==,!=)
    Ans- true / false
    Greater than (>)
    console.log(10>5)
    Output: true

    Smaller than (<)
    console.log(10<20)
    Output: true

    Equal to (==)
    console.log(10==10)
    Output: true

    Not equal to (!=)
    console.log(10!=20)
    Output: true

    Greater than and equal to (>=)
    console.log(10>=20)
    Output: false

    Smaller than and equal to (<=)
    console.log(10<=20)
    Output: true

    console.log(10<=10)
    Output: true
  1. Logical operator (and, or, not)
    Ans- true / false
    true- non zero- -10,-2,9,"sparsh"= truthy value
    false- zero- 0
    and (&&)- Multiplication
    console.log(true && true);
    Output: true

    console.log(true && false);
    Output: false

    console.log(false && true);
    Output: false

    console.log(false && false);
    Output: false

    console.log(0 && 1);
    Output: 0 (0*1=0 false)

    console.log(1 && 1);
    Output: 1 (1*1=1 true)

    or ( || )- jodna or add
    console.log(true || true);
    Output: true (1+1=2=non zero = true)

    console.log(true || false);
    Output: true (1+0=1=non zero = true)

    console.log(false || true);
    Output: true (0+1=1=non zero = true)

    console.log(false || false);
    Output: false (0+0=0=zero = false)

    not (!)
    console.log(!true);
    Output: false

    console.log(!false);
    Output: true

    console.log(!10);
    Output: false

    console.log(!0);
    Output: true

    console.log(!-2);
    Output: false

    console.log(!"sparsh");
    Output: false
