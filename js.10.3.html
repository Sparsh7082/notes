                    function checkprime(num){
                        let nof=0;
                        for(let div=2;div*div<=num;div++){
                            if(num%div==0){
                                nof++;
                                break;
                            }
                        }
                        if(nof>0){
                            return "not prime"
                        } else{
                            return "prime"
                        }
                    }
                    let rv = checkprime(11);
