Arrays Traversal

work something on each element of array

Multiplying each element by 2

                
                    let arr = [1,2,3,4,5]

                    for(let i=0;i<arr.length;i++){
                        console.log(arr[i]*2)
                    }

Output:

                2
                4
                6
                8
                10

Print even numbers from an array

                
                    let arr = [1,2,3,4,5]

                    for(let i=0;i<arr.length;i++){
                        if(arr[i]%2==0){
                            console.log(arr[i]);
                        }
                    }

Output:

                2
                4

Print maximum number from an array

                
                    let arr = [1,2,3,4,5]
                    
                    let max = 0;
                    for(let i=0; i<arr.length;i++){
                        if(max<arr[i]){
                            max=arr[i]
                        }
                    }
                    console.log(max);

Output:

                5

Find first index of a number in array

                
                    let arr2 = [1,200,3,4,5,1,2,3,4,1,3]

                    let target = 1;
                    let ans = 0;
                    for (let i=0; i<arr2.length;i++){
                        if(target==arr2[i]){
                            ans = i;
                            break;
                        }
                    }
                    console.log(ans);

Output:

                0

Reverse an array

                
                    let arr = [1,2,3,4,5]

                    for(let i=arr.length-1;i>=0;i--){
                        console.log(arr[i]);   
                    }

Output:

                5
                4
                3
                2
                1
