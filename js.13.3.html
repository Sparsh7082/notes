let arr = [100,200,"vikas",[1,2,3,4],function fun(){console.log("I am fun");},function fun2(){return "dice academy"}]

For loop

                for(let i=0;i<arr.length;i++){
                    console.log(arr[i])
                }

Output:

                100
                200
                vikas
                [ 1, 2, 3, 4 ]
                [Function: fun]
                [Function: fun2]

While loop

                let i=0;
                while(i<arr.length){
                    console.log(arr[i]);
                    i++;    
                }

Output:

                100
                200
                vikas
                [ 1, 2, 3, 4 ]
                [Function: fun]
                [Function: fun2]

Do While loop

                let i=0;

                    console.log(arr[i]);

                200

                [Function: fun]
